CREATE DATABASE IF NOT EXISTS SML_BD
DEFAULT CHARACTER SET utf8
DEFAULT COLLATE utf8_general_ci;


CREATE TABLE IF NOT EXISTS PRESTADOR (
    ID_PREST INT NOT NULL AUTO_INCREMENT,
    NOME VARCHAR (40) NOT NULL,
    CPF CHAR(11) NOT NULL UNIQUE,
    N_CONSELHO CHAR (5) NOT NULL UNIQUE,
    ESPECIALIDADES VARCHAR (40) NOT NULL,
    MAIL VARCHAR (40) NOT NULL,
    TEL_1 VARCHAR (11) NOT NULL,
    TEL_2 VARCHAR (11),
    CONSULTORIO VARCHAR (40),
    FIL_MAE VARCHAR (40) NOT NULL,
    FIL_PAI VARCHAR (40),
    INF_ADC VARCHAR (100),
    PRIMARY KEY (ID_PREST)
)DEFAULT CHARSET utf8;
